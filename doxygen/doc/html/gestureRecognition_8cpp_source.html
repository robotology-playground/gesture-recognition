384 void GestRecognition::computeContrastandOrientation(IplImage* img, IplImage* arctan, IplImage* contrast, double contrastTh, int aperture)
385 {
386  double max =0;
387  double min =0;
388  IplImage *img_t = cvCreateImage(cvGetSize(img),img->depth,1);
389  IplImage *img_f = cvCreateImage(cvGetSize(img),32,1);
390 
391  IplImage * derivativeX= cvCreateImage(cvGetSize(img),32,1);
392  IplImage * derivativeY= cvCreateImage(cvGetSize(img),32,1);
393 
394  if(img->nChannels>1)
395  cvCvtColor(img,img_t,CV_RGB2GRAY);
396  else
397  cvCopy(img,img_t,0);
398 
399  cvMinMaxLoc(img_t,&min,&max,NULL,NULL,NULL);
400  cvConvertScale(img_t,img_f,1.0/max,0);
401 
402  cvSobel(img_f,derivativeX,1,0,aperture);
403  cvSobel(img_f,derivativeY,0,1,aperture);
404 
405  cvZero(arctan);
406  cvZero(contrast);
407  cvCartToPolar(derivativeX, derivativeY, contrast, arctan, 0);
408 
409  cvThreshold(contrast,contrast,contrastTh,1,CV_THRESH_BINARY);
410 
411  cvReleaseImage(&img_f);
412  cvReleaseImage(&derivativeX);
413  cvReleaseImage(&derivativeY);
414  cvReleaseImage(&img_t);
415 }

430 void GestRecognition::computeFlowDense(IplImage* previous, IplImage* current, Mat &optFlow)
431 {
432  Mat pBgr=cvarrToMat(previous);
433  Mat cBgr=cvarrToMat(current);
434  Mat pGray(pBgr.rows,pBgr.cols,CV_8UC1);
435  Mat cGray(cBgr.rows,cBgr.cols,CV_8UC1);
436 
437  cvtColor(pBgr,pGray,CV_BGR2GRAY);
438  cvtColor(cBgr,cGray,CV_BGR2GRAY);
439  calcOpticalFlowFarneback(pGray,cGray,optFlow,0.5,1,3,15,5,1.1,OPTFLOW_FARNEBACK_GAUSSIAN);
440 
441  flow2D.clear();
442  yarp::sig::Vector v(4);
443  int scale=((optFlow.rows==480)&&(optFlow.cols==640))?2:1;
444  for (int i=0; i<previous->height; i++)
445  {
446  for (int j=0; j<previous->width; j++)
447  {
448  v[0]=(float)j/scale; // prev X
449  v[1]=(float)i/scale; // prev Y
450  v[2]=optFlow.ptr<float>(i)[2*j]/scale; // delta X
451  v[3]=optFlow.ptr<float>(i)[2*j+1]/scale; // delta Y
452 
453  if (norm(v.subVector(2,3))>thModFlow)
454  flow2D.push_back(v);
455  else
456  optFlow.ptr<float>(i)[2*j]=optFlow.ptr<float>(i)[2*j+1]=0.0F;
457  }
458  }
459 }

461 void GestRecognition::computeFlowHistogram()
462 {
463  flowHist.clear();
464 
465  if (flow3D.size()==0)
466  {
467  for (int i=0; i<dimension*dimension*dimension; i++)
468  flowHist.push_back(0);
469  return;
470  }
471 
472  vector<Mat> matHistogram;
473  for (int i=0; i<dimension; i++)
474  {
475  Mat matrix=Mat::zeros(dimension,dimension,CV_32F);
476  matHistogram.push_back(matrix);
477  }
478 
479  for (unsigned int i=0; i<flow3D.size(); i++)
480  {
481  int x,y,z;
482  double n=sqrt((flow3D.at(i).x*flow3D.at(i).x)+(flow3D.at(i).y*flow3D.at(i).y)+(flow3D.at(i).z*flow3D.at(i).z));
483  generateMatIndex((flow3D.at(i).x/n),x);
484  generateMatIndex((flow3D.at(i).y/n),y);
485  generateMatIndex((flow3D.at(i).z/n),z);
486  matHistogram.at(x).ptr<float>(y)[z]+=1;
487  }
488 
489  double normalizer=0;
490  for (unsigned int i=0; i<matHistogram.size(); i++)
491  {
492  for (int j=0; j<dimension; j++)
493  {
494  for (int k=0; k<dimension; k++)
495  {
496  flowHist.push_back(matHistogram.at(i).ptr<float>(j)[k]);
497  normalizer=normalizer+matHistogram.at(i).ptr<float>(j)[k];
498  }
499  }
500  }
501 
502  for (unsigned int i=0; i<flowHist.size(); i++)
503  flowHist[i]=flowHist[i]/normalizer;
504 }

544 void GestRecognition::computeSceneFlow(IplImage* previousMask,IplImage* currentMask, Mat &optFlow)
545 {
546  flow3D.clear();
547  Mat maskTempO=cvarrToMat(previousMask);
548 
549  Mat maskTempN=cvarrToMat(currentMask);
550  yarp::sig::Vector oldPoint(3);
551  yarp::sig::Vector currPoint(3);
552  Point3f flow;
553  for(unsigned int i=0; i<flow2D.size(); i++)
554  {
555  float x,y,deltaX,deltaY=0;
556  x=flow2D[i][0];
557  y=flow2D[i][1];
558  deltaX=flow2D[i][2];
559  deltaY=flow2D[i][3];
560 
561  if((cvRound(y+deltaY))<0 || (cvRound(x+deltaX))<0 || (cvRound(y+deltaY))>=currentMask->height || (cvRound(x+deltaX))>=currentMask->width )
562  {
563  optFlow.ptr<float>(cvRound(y))[2*cvRound(x)]=0;
564  optFlow.ptr<float>(cvRound(y))[2*cvRound(x)+1]=0;
565  continue;
566  }
567 
568  if(cvRound(y)<0 || cvRound(x)<0 || cvRound(y)>=previousMask->height || cvRound(x)>=previousMask->width)
569  {
570  optFlow.ptr<float>(cvRound(y))[2*cvRound(x)]=0;
571  optFlow.ptr<float>(cvRound(y))[2*cvRound(x)+1]=0;
572  continue;
573  }
574  float valOld= maskTempO.ptr<float>(cvRound(y))[cvRound(x)];
575  float valNew= maskTempN.ptr<float>(cvRound(y+deltaY))[cvRound(x+deltaX)];
576 
577  if (valOld==0 || valNew==0)
578  {
579  optFlow.ptr<float>(cvRound(y))[2*cvRound(x)]=0;
580  optFlow.ptr<float>(cvRound(y))[2*cvRound(x)+1]=0;
581  continue;
582  }
583 
584  retrieve3DPoint(previousMask,(int)x,(int)y, oldPoint);
585  retrieve3DPoint(currentMask,(int)(x+deltaX),(int)(y+deltaY),currPoint);
586 
587  flow.x=currPoint[0]-oldPoint[0];
588  flow.y=currPoint[1]-oldPoint[1];
589  flow.z=currPoint[2]-oldPoint[2];
590 
591  if(flow.x>0.001 || flow.y>0.001 || flow.z>0.001)
592  flow3D.push_back(flow);
593  }
594 }

596 void GestRecognition::drawMotionField(IplImage* imgMotion, int xSpace, int ySpace, float cutoff, float multiplier, CvScalar color)
597 {
598  CvPoint p0 = cvPoint(0,0);
599  CvPoint p1 = cvPoint(0,0);
600  float deltaX, deltaY, angle, hyp;
601 
602  for(unsigned int i=0; i<flow2D.size(); i++)
603  {
604  p0.x = (int)flow2D[i][0];
605  p0.y = (int)flow2D[i][1];
606  deltaX = flow2D[i][2];
607  deltaY = -(flow2D[i][3]);
608  angle = atan2(deltaY, deltaX);
609  hyp = sqrt(deltaX*deltaX + deltaY*deltaY);
610 
611  if(hyp > cutoff)
612  {
613  p1.x = p0.x + cvRound(multiplier*hyp*cos(angle));
614  p1.y = p0.y + cvRound(multiplier*hyp*sin(angle));
615  cvLine( imgMotion, p0, p1, color,1, CV_AA, 0);
616  p0.x = p1.x + cvRound(3*cos(angle-CV_PI + CV_PI/4));
617  p0.y = p1.y + cvRound(3*sin(angle-CV_PI + CV_PI/4));
618  cvLine( imgMotion, p0, p1, color,1, CV_AA, 0);
619 
620  p0.x = p1.x + cvRound(3*cos(angle-CV_PI - CV_PI/4));
621  p0.y = p1.y + cvRound(3*sin(angle-CV_PI - CV_PI/4));
622  cvLine( imgMotion, p0, p1, color,1, CV_AA, 0);
623  }
624  }

627 inline string GestRecognition::findNearest(yarp::sig::Vector &currPoint, Skeleton& jointsMap)
628 {
629  double minDist = 100;
630  string index="";
631 
632  double x1=currPoint[0];
633  double y1=currPoint[1];
634  double z1=currPoint[2];
635 
636  Skeleton::iterator iterator;
637  for (iterator=jointsMap.begin(); iterator!=jointsMap.end(); iterator++)
638  {
639  double x2=iterator->second.x;
640  double y2=iterator->second.y;
641  double z2=iterator->second.z;
642 
643  double dist=(x1-x2)*(x1-x2);
644  dist=dist+(y1-y2)*(y1-y2);
645  dist=dist+(z1-z2)*(z1-z2);
646 
647  dist=sqrt(dist);
648 
649  if(dist<minDist)
650  {
651  minDist=dist;
652  index=iterator->first;
653  }
654  }
655 
656  return index;
657 }

748 void GestRecognition::thresholdBW(IplImage *img, IplImage* res, float distance, float metersTh)
749 {
750  ushort mmdist=(ushort) (distance*1000);
751  ushort mmth=(ushort) (metersTh*1000);
752  mmdist=mmdist<<3;
753  mmth=mmth<<3;
754  distance=((1.0*mmdist/0xFFF8));
755  metersTh=((1.0*mmth/0xFFF8));
756  CvSize size =cvGetSize(img);
757 
758  IplImage *up= cvCreateImage(size,img->depth,1);
759  IplImage * down=cvCreateImage(size,img->depth,1);
760  IplImage * mask=cvCreateImage(size,IPL_DEPTH_8U,1);
761 
762  cvThreshold(img, up, distance+metersTh, 255, CV_THRESH_TOZERO_INV);
763  cvThreshold(img, down, distance-metersTh, 255, CV_THRESH_TOZERO);
764 
765  cvConvertScale(res,mask,255,0);
766  cvThreshold(mask,mask,0,255,CV_THRESH_BINARY);

768  cvAnd(up,down,res,mask);

770  cvReleaseImage(&up);
771  cvReleaseImage(&down);
772  cvReleaseImage(&mask);

791 
792  Point2f ws;
793  ws.x=wrist.x-shoulder.x;
794  ws.y=wrist.z-shoulder.z;
795 
796  double wsy=wrist.y-shoulder.y;
797

799  es.x=elbow.x-shoulder.x;
800  es.y=elbow.z-shoulder.z;

804  double n1=norm(we);
805  double n2=norm(ws);
806  double n3=norm(es);
807 
808  if((n1>0.30 || n2>0.30 || n3>0.30) || (wey>0.0 || wsy>0.0 || esy>0.0) || fabs(wey)<0.15 || fabs(wsy)<0.15 || fabs(esy)<0.15)

235 void SceneFlowThread::drawFlowModule(IplImage* imgMotion)
236 {
237  IplImage * module =cvCreateImage(cvSize(imgMotion->width,imgMotion->height),32,1);
238  IplImage * moduleU =cvCreateImage(cvSize(imgMotion->width,imgMotion->height),8,1);
239  Mat vel[2];
240  split(optFlow,vel);
241  IplImage tx=(Mat)vel[0];
242  IplImage ty=(Mat)vel[1];
243 
244  IplImage* velxpow=cvCloneImage(&tx);
245  IplImage* velypow=cvCloneImage(&ty);
246 
247  cvPow(&tx, velxpow, 2);
248  cvPow(&ty, velypow, 2);
249 
250  cvAdd(velxpow, velypow, module, NULL);
251  cvPow(module, module, 0.5);
252  cvNormalize(module, module, 0.0, 1.0, CV_MINMAX, NULL);
253  cvZero(imgMotion);
254  cvConvertScale(module,moduleU,255,0);
255  cvMerge(moduleU,moduleU,moduleU,NULL,imgMotion);
256  cvReleaseImage(&module);
257  cvReleaseImage(&moduleU);
258 }

305 void SceneFlowThread::computeFlowHistogram(const vector<Point3f>& flow3D, yarp::sig::Vector& flowHist)
306 {
307  flowHist.clear();
308 
309  if (flow3D.size()==0)
310  {
311  for (int i=0; i<dimension*dimension*dimension; i++)
312  flowHist.push_back(0);
313  return;
314  }
315 
316  vector<Mat> matHistogram;
317  for (int i=0; i<dimension; i++)
318  {
319  Mat matrix=Mat::zeros(dimension,dimension,CV_32F);
320  matHistogram.push_back(matrix);
321  }
322 
323  for (unsigned int i=0; i<flow3D.size(); i++)
324  {
325  int x,y,z;
326  double n=sqrt((flow3D.at(i).x*flow3D.at(i).x)+(flow3D.at(i).y*flow3D.at(i).y)+(flow3D.at(i).z*flow3D.at(i).z));
327  generateMatIndex((flow3D.at(i).x/n),x);
328  generateMatIndex((flow3D.at(i).y/n),y);
329  generateMatIndex((flow3D.at(i).z/n),z);
330  if(x<0 || y<0 || z<0 || x>=dimension || y>=dimension || z>=dimension)
331  continue;
332  matHistogram.at(x).ptr<float>(y)[z]+=1;
333  }
334 
335  double normalizer=0;
336  for (unsigned int i=0; i<matHistogram.size(); i++)
337  {
338  for (int j=0; j<dimension; j++)
339  {
340  for (int k=0; k<dimension; k++)
341  {
342  flowHist.push_back(matHistogram.at(i).ptr<float>(j)[k]);
343  normalizer=normalizer+matHistogram.at(i).ptr<float>(j)[k];
344  }
345  }
346  }
347 
348  for (unsigned int i=0; i<flowHist.size(); i++)
349  {
350  if(normalizer>0)
351  flowHist[i]=flowHist[i]/normalizer;
352  }
353 
354  return;
355 }
